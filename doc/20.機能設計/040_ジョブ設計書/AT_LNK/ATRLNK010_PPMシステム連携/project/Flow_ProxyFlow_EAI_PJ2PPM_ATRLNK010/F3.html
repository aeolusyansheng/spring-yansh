with LAYER_ORGANS(BASE_HR_ORGAN_ID, DEPT_CD, DEPT_SNM, HIGH_HR_ORGAN_ID, ORGAN_LAYER_SEG) as (
select BASE.HR_ORGAN_ID,
BASE.DEPT_CD,
BASE.DEPT_SNM,
BASE.HIGH_HR_ORGAN_ID,
BASE.ORGAN_LAYER_SEG
from PJTM_PJ_ORGAN BASE
inner join PJTM_PJ_HNYO_SEG HNYO_SEG
on BASE.DEPT_CD = HNYO_SEG.HNYO_SEG_DT_CD
and HNYO_SEG.HNYO_SEG_CD = '076'
where
BASE.ORGAN_LAYER_SEG >= '20' /*事業本部階層*/
and BASE.DEL_FLG = '0'
union all
select
CHILD_ORG.HR_ORGAN_ID,
CHILD_ORG.DEPT_CD,
CHILD_ORG.DEPT_SNM,
CHILD_ORG.HIGH_HR_ORGAN_ID,
CHILD_ORG.ORGAN_LAYER_SEG
from LAYER_ORGANS PARENT_ORG
inner join PJTM_PJ_ORGAN CHILD_ORG
on PARENT_ORG.BASE_HR_ORGAN_ID = CHILD_ORG.HIGH_HR_ORGAN_ID
where
CHILD_ORG.ORGAN_LAYER_SEG >= '20'/*事業本部*/
and CHILD_ORG.DEL_FLG = '0'
)
select
OPE_EMP_INFO.DEPT_CD as EMP_DEPT_CD
, OPE_EMP_INFO.DEPT_SNM as EMP_DEPT
, EMP.EMP_CD as EMP_CD
, EMP.NAME as EMP_NM
, STATS.HNYO_SEG_DT_NM as STATS
, to_char(OPE_DT.OPE_DT,'YYYY/MM/DD') as OPE_DT
, nvl2(PRJ.SI_DEAL_CD,PRJ.SI_DEAL_CD,'-') as SI_DEAL_CD
, nvl2(PRJ.SI_DEAL_CD,SI.SI_DEAL_NM,'-') as SI_DEAL_NM
, PRJ.PRJ_CD as PRJ_CD
, PRJ.PRJ_NM as PRJ_NM
, nvl2(PRJ.SI_DEAL_CD,WBS.WBS_CD,'-') as WBS_CD
, nvl2(PRJ.SI_DEAL_CD,WBS.WBS_NM,'-') as WBS_NM
, (
select OPE_ANA_ITEM_CD from PJTM_MI_OPERATING_ANA_ITEM where OPE_DT.OPE_DT_ID = OPE_ANA_ITEM_ID
) as OPE_DT_CD
, (
select OPE_ANA_ITEM_NM from PJTM_MI_OPERATING_ANA_ITEM where OPE_DT.OPE_DT_ID = OPE_ANA_ITEM_ID
) as OPE_DT_NM
, TSK.OPE_ANA_ITEM_CD_LG as OPE_ANA_ITEM_CD --稼働分析項目コード
, TSK.OPE_ANA_ITEM_NM_LG as OPE_ANA_ITEM_NM --稼働分析項目
, OPE_DT.OPE_DT_CMT as OPE_DT_CMT
, PRJ.PRJ_DEV_DEPT_CD as DEV_DEPT_CD --開発担当部課コード
, (
select DEPT_NM from PJTM_PJ_ORGAN where DEPT_CD = PRJ.PRJ_DEV_DEPT_CD
) as DEV_DEPT_NM --開発担当部課名
, ORG_E.DEPT_CD as EGY_DEPT_CD
, ORG_E.DEPT_SNM as EGY_DEPT
, EMP_E.EMP_CD as EGY_EMP_CD
, EMP_E.NAME as EGY_EMP
, PRJ.PRJ_CUST_CD as COST_CD
, CUST.CUST_SNM as COST_NM
, OPE_DT.OPE_TM as OPE_TM
, (
select PRJ_TYP_NM from PJTM_PJ_PRJ_TYP where PRJ_TYP_CD = PRJ.PRJ_TYP_CD
) as PRJ_TYP_NM --プロジェクトタイプ名
, TSK.TSK_ID as TSK_UID
from
PJTR_MI_OPE_DT OPE_DT
inner join (
select OPE_INFO.OPE_DT
, ASS.EMP_ID
, ORG.HR_ORGAN_ID
, ORG.DEPT_CD
, ORG.DEPT_SNM
, OPE_INFO.OPE_STATS
from PJTR_MI_OPE_DT_INFO OPE_INFO
inner join PJTM_PJ_ASSIGN ASS
on ASS.EMP_ID = OPE_INFO.ENT_PSN_ID
and ASS.COMPE_FLG = 1
and OPE_INFO.OPE_DT between ASS.VLD_STR_DT and ASS.VLD_END_DT
inner join PJTM_PJ_ORGAN ORG
on ORG.HR_ORGAN_ID = ASS.HR_ORGAN_ID
) OPE_EMP_INFO
on OPE_EMP_INFO.EMP_ID = OPE_DT.ENT_PSN_ID
and OPE_EMP_INFO.OPE_DT = OPE_DT.OPE_DT
inner join PJTM_PJ_EMP EMP
on EMP.EMP_ID = OPE_DT.ENT_PSN_ID
and (SELECT BUSI_DT FROM PJTM_FW_BUSI_DT WHERE ON_BAT_SEG = 'OL') between EMP.VLD_STR_DT and EMP.VLD_END_DT
inner join PJTM_PJ_HNYO_SEG STATS
on STATS.HNYO_SEG_CD = '075'
and stats.HNYO_SEG_DT_CD = OPE_EMP_INFO.OPE_STATS
inner join PJVR_MI_PRJ_UNI PRJ
on PRJ.PRJ_CD = OPE_DT.PRJ_CD
inner join PJVR_MI_TSK_UNI TSK
on TSK.TSK_MK_SYS_SEG = OPE_DT.TSK_MK_SYS_SEG
and TSK.TSK_ID = OPE_DT.TSK_ID

on SI.SI_DEAL_CD = PRJ.SI_DEAL_CD
left join PJTR_PJ_WBS WBS
on WBS.WBS_ID = TSK.WBS_ID
left join (SELECT *
FROM PJTM_PJ_EMP
WHERE (SELECT BUSI_DT FROM PJTM_FW_BUSI_DT WHERE ON_BAT_SEG = 'OL') >= VLD_STR_DT
AND (SELECT BUSI_DT FROM PJTM_FW_BUSI_DT WHERE ON_BAT_SEG = 'OL') <= VLD_END_DT
) EMP_E
on EMP_E.EMP_ID = OPE_DT.EGY_CHG_ID
left join PJTM_PJ_ASSIGN ASS_E
on ASS_E.EMP_ID = EMP_E.EMP_ID
and ASS_E.COMPE_FLG = 1
and OPE_DT.OPE_DT between ASS_E.VLD_STR_DT and ASS_E.VLD_END_DT
left join PJTM_PJ_ORGAN ORG_E
on ORG_E.HR_ORGAN_ID = ASS_E.HR_ORGAN_ID
left join PJTM_PJ_CUST CUST
on CUST.CUST_CD = PRJ.PRJ_CUST_CD
where TO_CHAR(OPE_DT.OPE_DT,'YYYY/MM/DD') between CONCAT('$CAL_STR_MN$','/01')

order by EMP_DEPT_CD, EMP_CD, OPE_DT, PRJ_CD, WBS_NM
