with LAYER_ORGANS(BASE_HR_ORGAN_ID, DEPT_CD, DEPT_SNM, HIGH_HR_ORGAN_ID, ORGAN_LAYER_SEG) as (
select BASE.HR_ORGAN_ID,
BASE.DEPT_CD,
BASE.DEPT_SNM,
BASE.HIGH_HR_ORGAN_ID,
BASE.ORGAN_LAYER_SEG
from PJTM_PJ_ORGAN BASE
inner join PJTM_PJ_HNYO_SEG HNYO_SEG
on BASE.DEPT_CD = HNYO_SEG.HNYO_SEG_DT_CD
and HNYO_SEG.HNYO_SEG_CD = '076'
where
BASE.ORGAN_LAYER_SEG >= '20' /*事業本部階層*/
and BASE.DEL_FLG = '0'
union all
select
CHILD_ORG.HR_ORGAN_ID,
CHILD_ORG.DEPT_CD,
CHILD_ORG.DEPT_SNM,
CHILD_ORG.HIGH_HR_ORGAN_ID,
CHILD_ORG.ORGAN_LAYER_SEG
from LAYER_ORGANS PARENT_ORG
inner join PJTM_PJ_ORGAN CHILD_ORG
on PARENT_ORG.BASE_HR_ORGAN_ID = CHILD_ORG.HIGH_HR_ORGAN_ID
where
CHILD_ORG.ORGAN_LAYER_SEG >= '20'/*事業本部*/
and CHILD_ORG.DEL_FLG = '0'
)

SELECT PRJ.PRJ_CD
FROM PJTR_PJ_PRJ PRJ
INNER JOIN (SELECT A.PRJ_CD, MAX(TO_CHAR(A.UPD_TS,'YYYY/MM/DD HH24:MI:SS')) AS CHK_TS
FROM (SELECT WBS_TSK.PRJ_CD, wbs_tsk.UPD_TS --MAX(TO_CHAR(WBS_TSK.UPD_TS, 'YYYY/MM/DD HH24:MI:SS')) AS CHK_TS
FROM PJTR_PJ_WBS_TSK WBS_TSK
UNION
SELECT WBS.PRJ_CD, wbs.UPD_TS --MAX(TO_CHAR(WBS.UPD_TS, 'YYYY/MM/DD HH24:MI:SS'))
FROM PJTR_PJ_WBS WBS) A
GROUP BY A.PRJ_CD) TSK
ON TSK.PRJ_CD = PRJ.PRJ_CD
inner join PJTM_PJ_EMP EMP_EGY
on EMP_EGY.EMP_CD = PRJ.PRJ_EGY_EMP_CD
and (SELECT BUSI_DT FROM PJTM_FW_BUSI_DT WHERE ON_BAT_SEG = 'OL') >= EMP_EGY.VLD_STR_DT
and (SELECT BUSI_DT FROM PJTM_FW_BUSI_DT WHERE ON_BAT_SEG = 'OL') <= EMP_EGY.VLD_END_DT
inner join PJTM_PJ_ASSIGN EGY_ASS
on EGY_ASS.EMP_ID = EMP_EGY.EMP_ID
and EGY_ASS.COMPE_FLG = 1
and (SELECT BUSI_DT FROM PJTM_FW_BUSI_DT WHERE ON_BAT_SEG = 'OL') >= EGY_ASS.VLD_STR_DT
and (SELECT BUSI_DT FROM PJTM_FW_BUSI_DT WHERE ON_BAT_SEG = 'OL') <= EGY_ASS.VLD_END_DT
inner join PJTM_PJ_ORGAN ORG_EGY
on ORG_EGY.HR_ORGAN_ID = EGY_ASS.HR_ORGAN_ID
WHERE TSK.CHK_TS >= CONCAT
((SELECT NVL(TO_CHAR(DT_MST.MNG_DT,'yyyy/mm/dd'),'1900/01/01')
FROM PJTM_PJ_HNYO_DT_MNG_MST DT_MST
WHERE DT_MST.HNYO_SEG_CD='$SEG_CD$'
AND DT_MST.HNYO_SEG_DT_CD='$SEG_DT_CD$'),' 00:00:00')
AND (exists (SELECT * FROM LAYER_ORGANS WHERE LAYER_ORGANS.DEPT_CD = PRJ.PRJ_DEV_DEPT_CD)
or exists (SELECT * FROM LAYER_ORGANS WHERE LAYER_ORGANS.DEPT_CD = ORG_EGY.DEPT_CD)
or exists (SELECT * FROM LAYER_ORGANS WHERE LAYER_ORGANS.DEPT_CD = PRJ.EGY_TRNS_DEPT_CD))
AND
(PRJ.PRJ_STATS_SEG in （'030','040','050','060','080')
or exists(
select 'X' from PJTR_PJ_PROSC_EST PROSC
inner join PJTR_PJ_PRJ PRJ2
on PROSC.PRJ_CD = PRJ2.PRJ_CD
inner join PJTM_PJ_PRJ_TYP_SEG PRJ_TYP
on PRJ2.PRJ_TYP_CD = PRJ_TYP.PRJ_TYP_CD
and PRJ_TYP.CTGRY_SEG_CD = '009'
and PRJ_TYP.CTGRY_SEG_DT_CD = '015'
and PRJ_TYP.VLD_FLG = '1'
where PRJ2.PRJ_STATS_SEG = '010'
and PROSC.PROS_VLD_FLG = '1'
and PROSC.DEL_FLG = '0'
and PRJ2.PRJ_CD = PRJ.PRJ_CD
))
ORDER BY PRJ.PRJ_CD
